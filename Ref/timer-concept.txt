    <script>
        const HOLD_TIME = 10;
        const REST_TIME = 3;
        const TOTAL_REPS = 6;

        let timeLeft = HOLD_TIME;
        let currentRep = 1;
        let isHolding = true;
        let isRunning = false;
        let timerInterval = null;

        const timerEl = document.getElementById('timer');
        const sleeveEl = document.getElementById('sleeve');
        const startBtn = document.getElementById('start-btn');
        const resetBtn = document.getElementById('reset-btn');
        const phaseLabel = document.getElementById('phase-label');
        const repLabel = document.getElementById('rep-count');
        const appWrap = document.getElementById('app');
        const stageText = document.getElementById('stage-text');
        const timerText = document.getElementById('timer');

        function updateUI() {
            timerEl.innerText = Math.ceil(timeLeft).toString().padStart(2, '0');
            const total = isHolding ? HOLD_TIME : REST_TIME;
            const percentage = ((total - timeLeft) / total) * 100;
            sleeveEl.style.height = `${percentage}%`;
            
            repLabel.innerText = `${currentRep.toString().padStart(2, '0')} / ${TOTAL_REPS.toString().padStart(2, '0')}`;
            
            if (isHolding) {
                appWrap.classList.remove('state-rest');
                phaseLabel.innerText = 'HOLD';
                stageText.innerText = 'Tension Loading';
                document.documentElement.style.setProperty('--tension-hold', '#ff9d00');
                if (isRunning) timerText.classList.add('vibrate');
            } else {
                appWrap.classList.add('state-rest');
                phaseLabel.innerText = 'REST';
                stageText.innerText = 'System Release';
                timerText.classList.remove('vibrate');
            }
        }

        function toggleTimer() {
            if (isRunning) {
                clearInterval(timerInterval);
                startBtn.innerText = 'Resume';
                timerText.classList.remove('vibrate');
            } else {
                startBtn.innerText = 'Suspend';
                timerInterval = setInterval(() => {
                    timeLeft -= 0.1;
                    if (timeLeft <= 0) {
                        handlePhaseSwitch();
                    }
                    updateUI();
                }, 100);
            }
            isRunning = !isRunning;
        }

        function handlePhaseSwitch() {
            if (isHolding) {
                isHolding = false;
                timeLeft = REST_TIME;
            } else {
                isHolding = true;
                timeLeft = HOLD_TIME;
                currentRep++;
                if (currentRep > TOTAL_REPS) {
                    resetTimer();
                    alert("Sequence Complete. Maintain Neutral Spine.");
                }
            }
        }

        function resetTimer() {
            clearInterval(timerInterval);
            isRunning = false;
            isHolding = true;
            timeLeft = HOLD_TIME;
            currentRep = 1;
            startBtn.innerText = 'Initiate';
            timerText.classList.remove('vibrate');
            updateUI();
        }

        startBtn.addEventListener('click', toggleTimer);
        resetBtn.addEventListener('click', resetTimer);

        // Initial UI state
        updateUI();
    </script>
